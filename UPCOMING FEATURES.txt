 * Draws a settings gear icon using pure OpenGL geometry.
 * @param x Center X position
 * @param y Center Y position
 * @param radius The size of the gear
 * @param color The hex color (e.g., 0xFFFFFFFF)
 * @param rotation The rotation in degrees (pass System.currentTimeMillis() / 20.0 for spinning!)
 */
public void drawMathGear(float x, float y, float radius, int color, float rotation) {
    float innerRadius = radius * 0.6f; // The "hole" in the middle
    float toothLength = radius * 0.25f; // How far the teeth stick out
    int teethCount = 8;
    
    GlStateManager.pushMatrix();
    GlStateManager.translate(x, y, 0);
    GlStateManager.rotate(rotation, 0, 0, 1);
    GlStateManager.disableTexture2D();
    GlStateManager.enableBlend();
    
    // Set the color
    float f3 = (float)(color >> 24 & 255) / 255.0F;
    float f = (float)(color >> 16 & 255) / 255.0F;
    float f1 = (float)(color >> 8 & 255) / 255.0F;
    float f2 = (float)(color & 255) / 255.0F;
    GlStateManager.color(f, f1, f2, f3);

    // 1. Draw the Main "Ring" (Outer Body)
    GL11.glBegin(GL11.GL_QUAD_STRIP);
    for (int i = 0; i <= 360; i += 10) {
        float angle = (float) Math.toRadians(i);
        GL11.glVertex2f((float) Math.cos(angle) * innerRadius, (float) Math.sin(angle) * innerRadius);
        GL11.glVertex2f((float) Math.cos(angle) * radius, (float) Math.sin(angle) * radius);
    }
    GL11.glEnd();

    // 2. Draw the Teeth
    float toothWidth = 360f / (teethCount * 2);
    for (int i = 0; i < teethCount; i++) {
        float angle = i * (360f / teethCount);
        
        GL11.glBegin(GL11.GL_QUADS);
        // We define the 4 corners of each tooth based on the angle
        for (float step = -toothWidth/2; step <= toothWidth/2; step += toothWidth) {
            float a1 = (float) Math.toRadians(angle - toothWidth/2);
            float a2 = (float) Math.toRadians(angle + toothWidth/2);
            
            GL11.glVertex2f((float) Math.cos(a1) * radius, (float) Math.sin(a1) * radius);
            GL11.glVertex2f((float) Math.cos(a1) * (radius + toothLength), (float) Math.sin(a1) * (radius + toothLength));
            GL11.glVertex2f((float) Math.cos(a2) * (radius + toothLength), (float) Math.sin(a2) * (radius + toothLength));
            GL11.glVertex2f((float) Math.cos(a2) * radius, (float) Math.sin(a2) * radius);
        }
        GL11.glEnd();
    }

    GlStateManager.enableTexture2D();
    GlStateManager.popMatrix();
